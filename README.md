CREATE OR REPLACE TABLE BIGMART_TRAIN(
ITEM_IDENTIFIER VARCHAR,
ITEM_WEIGHT NUMBER,
ITEM_FAT_CONTENT VARCHAR,
ITEM_VISIBILITY FLOAT,
ITEM_TYPE VARCHAR,
ITEM_MRP FLOAT,
OUTLET_IDENTIFIER VARCHAR,
OUTLET_ESTABLISHMENT_YEAR NUMBER,
OUTLET_SIZE VARCHAR,
OUTLET_LOCATION_TYPE VARCHAR,
OUTLET_TYPE VARCHAR,
ITEM_OUTLET_SALES FLOAT
);

CREATE OR REPLACE STAGE BIGMART_STAGE;

COPY INTO BIGMART_TRAIN
FROM @BIGMART_STAGE
FILE_FORMAT =(
TYPE = CSV
SKIP_HEADER =1
FIELD_OPTIONALLY_ENCLOSED_BY ='"'
)

SELECT * FROM ETL_PROJECT_DB.SALES_PIPELINE.BIGMART_TRAIN LIMIT 10;

CREATE OR REPLACE TABLE BIGMART_SILVER AS
SELECT
ITEM_IDENTIFIER,
COALESCE(ITEM_WEIGHT,0)AS ITEM_WEIGHT,
UPPER(ITEM_FAT_CONTENT) AS ITEM_FAT_CONTENT,
ITEM_VISIBILITY,
INITCAP(ITEM_TYPE) AS ITEM_TYPE,
ITEM_MRP,
OUTLET_IDENTIFIER,
OUTLET_ESTABLISHMENT_YEAR,
COALESCE(OUTLET_SIZE, 'UNKNOWN') AS OUTLET_SIZE,
UPPER(OUTLET_LOCATION_TYPE) AS OUTLET_LOCATION_TYPE,
UPPER(OUTLET_TYPE) AS OUTLET_TYPE,
ITEM_OUTLET_SALES
FROM ETL_PROJECT_DB.SALES_PIPELINE.BIGMART_TRAIN;

SELECT * FROM ETL_PROJECT_DB.SALES_PIPELINE.BIGMART_SILVER LIMIT 10;

CREATE OR REPLACE TABLE BIGMART_GOLD AS
SELECT
OUTLET_TYPE,
ITEM_TYPE,
COUNT(*) AS TOTAL_PRODUCTS,
ROUND(AVG(ITEM_MRP),2) AS AVG_PRICE,
ROUND(SUM(ITEM_OUTLET_SALES), 2) AS TOTAL_SALES
FROM ETL_PROJECT_DB.SALES_PIPELINE.BIGMART_SILVER
GROUP BY OUTLET_TYPE,ITEM_TYPE;

ON TABLE ETL_PROJECT_DB.SALES_PIPELINE.BIGMART_TRAIN;

CREATE OR REPLACE TASK BIGMART_SILVER_TASK
WAREHOUSE = ETL_WH
SCHEDULE ='1 MINUTE'
AS
MERGE INTO BIGMAT_SILVER T
USING (
    SELECT
        ITEM_IDENTIFIER,
        COALESCE(ITEM_WEIGHT,0) AS ITEM_WEIGHT,
        UPPER(ITEM_FAT_CONTENT) AS ITEM_FAT_CONTENT,
        ITEM VISIBILITY,
        INITCAP(ITEM_TYPE) AS ITEM_TYPE,
        ITEM_MRP,
        OUTLET_IDENTIFIER,
        OUTLET_ESTABLISHMENT_YEAR,
        COALESCE(OUTLET_SIZE,'UNKOWN') AS OUTLET_SIZE,
        UPPER(OUTLET_LOCATION_TYPE) AS OUTLET_LOCATION_TYPE,
        UPPER(OUTLET_TYPE) AS OUTLET_TYPE,
        ITEM_OUTLET_SALES
        FROM ETL_PROJECT_DB.SALES_PIPELINE.BIGMART_STREAM
) S
ON T.ITEM_IDENTIFIER = S.ITEM_IDENTIFIER

WHEN MATCHED THEN
UPDATE SET
   T.ITEM_OUTLET_SALES = S.ITEM_OUTLET_SALES

WHEN NOT MATCHED THEN
INSERT VALUES (
S.ITEM_IDENTIFIER,
S.ITEM_WEIGHT,
S.ITEM_FAT_CONTENT,
S.ITEM_VISIBILITY,
S.ITEM_TYPE,
S.ITEM_MRP,
S.OUTLET_IDENTIFIER,
S.OUTLET_ESTABLISHMENT_YEAR,
S.OUTLET_SIZE,
S.OUTLET_LOCATION_TYPE,
S.OUTLET_TYPE,
S.ITEM_OUTLET_SALES
);

ALTER TASK BIGMART_SILVER_TASK RESUME;

SHOW TASKS;
